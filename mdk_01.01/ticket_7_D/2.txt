using System;

class SieveOfEratosthenes
{
    static void Sieve(int n)
    {
        bool[] prime = new bool[n + 1]; // создаем массив булевых значений для хранения информации о простых числах
        for (int i = 0; i <= n; i++)
        {
            prime[i] = true; // инициализируем все значения как простые числа
        }

        for (int p = 2; p * p <= n; p++)
        {
            if (prime[p] == true) // если число является простым
            {
                for (int i = p * p; i <= n; i += p)
                {
                    prime[i] = false; // отмечаем все кратные как не простые
                }
            }
        }

        // выводим простые числа
        Console.WriteLine("Простые числа от 1 до " + n + " это:");
        for (int p = 2; p <= n; p++)
        {
            if (prime[p] == true)
            {
                Console.Write(p + " ");
            }
        }
    }

    static void Main()
    {
        int n = 1000; // задаем предел для поиска простых чисел
        Sieve(n);
    }
}
